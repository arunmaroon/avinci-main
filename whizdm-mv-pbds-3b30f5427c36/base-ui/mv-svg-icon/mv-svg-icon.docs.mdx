import { Box } from "@mui/material";

A component that does something special and renders text in a div.

### Component usage

```js
import { MvSvgIcon } from "./mv-svg-icon";
import { IconNames } from "@mvloans/base-ui.common";
```

### Render SVG Icons!

```js live
<>
  {Object.keys(IconNames).map((iconName) => (
    <Box sx={{ display: "flex", alignItems: "center", justifyContent: "space-between", margin: "5px" }}>
      <MvSvgIcon name={iconName} />
      <span>{iconName}</span>
    </Box>
  ))}
</>
```
