import { TextField } from "./text-field";
import { ThemeProvider } from "@mvloans/base-ui.theme-provider";
import Typography from "@mvloans/base-ui.typography";
import Button from "@mui/material/Button";
import { Box } from "@mui/material";
import { TypographyVariants } from "@mvloans/base-ui.common";

TextField component

### Component usage

```js
<ThemeProvider>
  <TextField value="Hello World" />
</ThemeProvider>
```

### Render Skeleton

```js live
<ThemeProvider>
  <Box sx={{ width: { xs: 360, md: 312 } }}>
    <TextField isLoading label="Label Value" placeholder="Placeholder" helperText="Assistive Text" error={false} disabled={false} />
  </Box>
</ThemeProvider>
```

### Render Text Field

```js live
<ThemeProvider>
  <Box sx={{ width: { xs: 360, md: 312 } }}>
    <TextField label="Label Value" placeholder="Placeholder" helperText="Assistive Text" error={false} disabled={false} />
  </Box>
</ThemeProvider>
```

### Render Suffix Link

```js live
<ThemeProvider>
  <Box sx={{ width: { xs: 360, md: 312 } }}>
    <TextField
      showSuffix
      suffix={
        <Typography colorValue="#28292D" href="https://moneyview.in/" variant={TypographyVariants.bodySmallHighEmphasis}>
          Link
        </Typography>
      }
      label="Label Value"
      placeholder="Placeholder"
      helperText="Assistive Text"
      error={false}
      disabled={false}
    />
  </Box>
</ThemeProvider>
```
